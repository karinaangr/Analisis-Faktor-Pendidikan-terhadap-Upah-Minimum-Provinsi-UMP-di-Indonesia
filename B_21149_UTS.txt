# Overall Test
overall_p <- function(my_model) {
  f <- summary(my_model)$fstatistic
  p <- pf(f[1],f[2],f[3],lower.tail=F)
  attributes(p) <- NULL
  return(p)
}

overall_test = data.frame(matrix(ncol = 2, nrow = 6))
colnames(overall_test) = c("Model", "P.Value")
overall_test$Model = c("Model 1", "Model 2", "Model 3", "Model 4", "Model 5", "Model 6")
overall_test$P.Value = c(overall_p(model1),
                         overall_p(model2),
                         overall_p(model3),
                         overall_p(model4),
                         overall_p(model5),
                         overall_p(model6))
print(overall_test[,-5])
